    • High-accuracy classification of attack patterns

    Our Solution

    HybridML-IDS combines the temporal pattern recognition of LSTM networks with the ensemble learning of Random Forest and the high-dimensional classification power of SVM.

    Data Preprocessing Pipeline 

    class DataPreprocessor:
    def __init__(self):
        self.scaler = RobustScaler()
        self.transformer = PowerTransformer()
        self.feature_selector = SelectKBest(f_classif, k=20)
        
    def preprocess(self, X):
        # Handle missing values and infinite values
        X = X.replace([np.inf, -np.inf], np.nan).dropna()
        
        # Scale features robustly
        X_scaled = self.scaler.fit_transform(X)
        
        # Power transform for normality
        X_transformed = self.transformer.fit_transform(X_scaled)
        
        # Feature selection
        X_selected = self.feature_selector.fit_transform(X_transformed)
        
        return X_selected

    Hybrid Model Ensemble 

    class HybridIDSModel:
    def __init__(self):
        self.rf = RandomForestClassifier(n_estimators=200, max_depth=20)
        self.svm = SVC(kernel='rbf', C=1.0, probability=True)
        self.lstm = self._build_lstm()
        
    def predict(self, X):
        # Get individual model predictions
        rf_proba = self.rf.predict_proba(X)
        svm_proba = self.svm.predict_proba(X)
        lstm_proba = self.lstm.predict(X.reshape(1, -1))
        
        # Dynamic weighting based on F1 scores
        total_weight = self.rf_f1 + self.svm_f1 + self.lstm_f1
        hybrid_proba = (rf_proba*self.rf_f1 + 
                       svm_proba*self.svm_f1 + 
                       lstm_proba*self.lstm_f1) / total_weight
        
        return np.argmax(hybrid_proba, axis=1)
        
    def _build_lstm(self):
        model = Sequential([
            LSTM(256, input_shape=(None, 20), return_sequences=True),
            Dropout(0.3),
            LSTM(128),
            Dropout(0.3),
            Dense(64, activation='relu'),
            Dense(5, activation='softmax')
        ])
        model.compile(loss='categorical_crossentropy', optimizer='adam')
        return model
