class HybridIDSModel:
    def __init__(self):
        self.models = {
            'rf': RandomForestClassifier(n_estimators=200),
            'lstm': build_lstm_network(),
            'svm': SVC(kernel='rbf', probability=True)
        }
    
    def predict(self, X):
        # Dynamic weighting based on model confidence
        rf_proba = self.models['rf'].predict_proba(X)
        lstm_proba = self.models['lstm'].predict(X.reshape(1, -1))
        svm_proba = self.models['svm'].predict_proba(X)
        
        # Ensemble with precision-based weights
        hybrid_proba = (0.4*rf_proba + 0.3*lstm_proba + 0.3*svm_proba)
        return np.argmax(hybrid_proba, axis=1)
